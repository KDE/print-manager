    KCupsPrinter.cpp
    KCupsServer.cpp
    KCupsRequest.cpp
    PPDModel.cpp

    DevicesModel.cpp

    KIppRequest.h
    KCupsPasswordDialog.h
    KCupsConnection.h

    IDENTIFIER LIBKCUPS
    CATEGORY_NAME org.kde.plasma.printmanager.libkcups
    DESCRIPTION "Print Manager (libkcups)"
    EXPORT PMLOGS
)

install(TARGETS kcups ${KDE_INSTALL_TARGETS_DEFAULT_ARGS} LIBRARY NAMELINK_SKIP)
